44  template <std::size_t N, std::size_t BufferSize>

46 
47  array_of_string_view<N> result;
48 
49  // we change the status to false once we have processed an enumeration
50  // value, and we change it back to true once we find a separator (a comma
51  // ',')
52  bool status = true;
53 
54  std::size_t counter = 0u, fc = 0u, sc = 0u;
55 
56  while (sc < buffer.size()) {
57 
58  auto c = buffer[sc];
59 
60  if (c == ' ' || c == '\t' || c == '\n' || c == '\r' || c == '=') {
61  if (status) {
62  if (fc != sc) {
63  // we encountered a space after the name declaration
64  status = false;
65  result[counter++] = std::string_view(&buffer[fc], sc - fc);
66  }
67 
68  // we might be at the begining of the string
69  fc = ++sc;
70  } else
71  ++sc;
72  } else if (c == ',') {
73  // we reached the end of the declaration of an enumeration value
74 
75  if (status)
76  // we still must parse this value
77  result[counter++] = std::string_view(&buffer[fc], sc - fc);
78  else
79  status = true;
80 
81  fc = ++sc;
82  } else if (sc + 1 == buffer.size() && status)
83  result[counter++] = std::string_view(&buffer[fc], ++sc - fc);
84  else

86  }
